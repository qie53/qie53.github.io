springBoot整合shiro

  1. 需要的jar包

    1.         <dependency>
            <groupId>org.apache.shiro</groupId>
            <artifactId>shiro-spring-boot-starter</artifactId>
            <version>1.8.0</version>
        </dependency>

  2. 编写shiroConfig

    1. package com.edu.penguin.shirospringboot.config;

import org.apache.shiro.spring.web.ShiroFilterFactoryBean;
import org.apache.shiro.web.mgt.DefaultWebSecurityManager;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.LinkedHashMap;
import java.util.Map;

@Configuration
public class ShiroConfig{

    //创建realm 对象 1
    @Bean(name="userRealm")
    public UserRealm userRealm(){
        return new UserRealm();
    }

    //DefaultWebSecurityMangager 2
    @Bean(name="defaultWebSecurityManager")
    public DefaultWebSecurityManager defaultWebSecurityManager(@Qualifier("userRealm") UserRealm userRealm){
        DefaultWebSecurityManager securityManager = new DefaultWebSecurityManager();
        securityManager.setRealm(userRealm);
        return securityManager;
    }

    //shiroFilterFactoryBean 3
    @Bean(name="shiroFilterFactoryBean")
    public ShiroFilterFactoryBean shiroFilterFactoryBean(@Qualifier("defaultWebSecurityManager") DefaultWebSecurityManager defaultWebSecurityManager){
        ShiroFilterFactoryBean bean= new ShiroFilterFactoryBean();
        bean.setSecurityManager(defaultWebSecurityManager);

        //设置登录页面
//        bean.setLoginUrl("/login");
//
//        Map<String, String> filterMap = new LinkedHashMap<>();
//        /**
//         * anon 无需认证即可访问
//         * authc 必须认证才能使用
//         * user 用户拦截器，用户已经身份验证/记住我登录的都可
//         * perms 拥有对某个资源的权限才能访问
//         * role 拥有某个角色权限才能访问
//         * */
//        filterMap.put("/add","anon");
//        filterMap.put("/update","authc");
//
//        //添加shiro的内置过滤器
//        bean.setFilterChainDefinitionMap(filterMap);

        return bean;
    }
}

<!--￼9-->

  • @Autowired：自动装配，通过类型，名字

    • 如果不能唯一自动装配，则需要通过@Qualifier(value = “name”)

  • @ConfigurationProperties：如果你使用properties或yml进行注入就需要在想要注入的类上面写上 @ConfigurationProperties(prefix=”cat”) prefix是指定配置文件中的对应的注入

  • @PropertySource：加载指定的配置文件 用法

    • @PropertySource(value = “classpath:文件名会从resource出发”)

  • @Validated：JSR303校验 进行各种校验要导入包

    • <dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-validation</artifactId>
</dependency>

    • @Email()

    • @NotNull()：注入不能为空

    • @Max(value = 100,message = “最大不超过100，提示消息”)：最大不超过100

    • @Min(0)：最小不低于0

  • @responseBody：视图解析器不生效

  • @RestController：不解析视图的controller
